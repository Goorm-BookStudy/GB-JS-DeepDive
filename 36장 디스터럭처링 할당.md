# 36장 디스터럭처링 할당
- 구조화된 배열과 같은 이터러블 or 객체를 destructuring 하여 1개 이상의 변수에 개별적으로 할당하는 것

## 배열 디스트럭처링 할당

```
// ES5
var arr = [1,2,3];

var one = arr[0];
var two = arr[1];
var three = arr [2];

console.log(one, two, three); // 1 2 3

// ES6
const arr = [1,2,3];

const [one, two, thtee] = arr;

console.log(one, two, three); // 1 2 3


// 우변에 이터러블 할당 안 하면 에러 발생
const [x, y]; // SyntaxError
const [a, b]; = {}; // TypeError

// 배열 디스트럭처링 할당의 변수 선언문 - 선언과 할당 분리 but 권장하지 않음
// const 키워드로 변수선언을 할 수 없음
let x, y;
[x, y] = [1,2];

// 변수의 개수와 이터러블의 요소 개수가 반드시 일치할 필요 x
const [a,b] = [1,2];
console.log(a, b); // 1 2

const [c, d] = [1];
console.log(c, d); // 1 undefined

const [e, f] = [1,2,3];
console.log(e, f); // 1 2

// 디스트럭처링 할당을 위해 변수 기본값 설정 가능
const [a, b, c = 3] = [1,2];
console.log(a, b, c); // 1 2 3

// 기본값 보다 할당을 우선한다.
const [e, f = 10, g = 3] = [1,2];
console.log(e, f, g); // 1 2 3

// 배열 디스트럭처링 할당을 위한 변수에 Rest 파라미터와 유사하게 Rest 요소 ... 을 사용 가능
// Rest 요소는 마지막에 위치해야 한다.
const [x, ... y] = [1,2,3];
console.log(x, y); // 1, [2,3]
```

## 객체 디스트럭처링 할당

```jsx
// ES5
var user = {firstName: 'hamin', lastName: 'park' };

var firstName = user.firstName;
var lastName = user.lastName;

console.log(firstName, lastName); // hamin park

ES6
// 객체의 각 프로퍼티를 객체로부터 추출하여 1개 이상의 변수에 할당
// 우변에서 순서는 의미 x, 선언된 변수 이름과 프로퍼티 키가 일치하면 할당
const user = {firstName: 'hamin', lastName: 'park' };

// 프로퍼티 축약된 상태
const {lastName, firstName} = user;

// 축약 전 형태
const {lastName: lastName, firstName: firstName} = user;

// 프로퍼티 키와 다른 변수 이름으로 할당받으려면 
const {lastName: ln, firstName: fn} = user;
console.log(fn, ln);

console.log(firstName, lastName); // hamin park

// 배열 디스트럭처링과 동일하게 할당 연산자 왼쪽에 프로퍼티 값을 할당 받을 변수로 설정해야함
// 객체 리터럴 형태로
const user = {firstName, lastName} = {firstName: 'hamin', lastName: 'park' };

// 우변 객체 또는 객체로 평가될 수 있는 표현식을 할당하지 않으면 에러 발생
const {lastName, firstName}; // SyntaxError
const {lastName, firstName}; // TypeError

// 기본 값 설정
const { fistName = 'hamin', lastName} = { lastName: 'Lee'};
console.log(firstName, lastName); // hamin Lee

const {firstName: fn = 'hamin', lastName: ln} = { lastName: 'Lee'};
console.log(fn, ln); // hamin Lee

// 필요한 프로퍼티 값만 추출할 때 유용하다
cosnt str = 'hamin';

const {length} = str;
console.log(length); // 5

const todo = {id:1, content:'html'}
const {id} = todo;
console.log(id); // 1

// 객체를 인수로 전달받는 함수의 매개변수에서도 사용 가능
function printTodo(todo){
	console.log(`할일 ${todo.content}은 ${todo.completed ? '완료' : '비완료'} 상태입니					다.`);
}
printTodo({id: 1, content: 'html', completed: true}); // 할일 html은 완료 상태입니다.


// 더 좋은 가독성
function printTodo({content, completed}){
	console.log(`할일 ${content}은 ${completed ? '완료' : '비완료'} 상태입니다.`);
}
printTodo({id: 1, content: 'html', completed: true}); // 할일 html은 완료 상태입니다.

// 배열의 요소가 객체인 경우 혼용을 조심하자
const todo = [
  {id: 1, content: 'css', completed: true}
];

const [, {id}] =todos;

// 중첩 객체의 경우
const user = {
	name: 'hamin'
  	address: {
  		city: 'seoul'
	}
};

const {address: {city}} = user;
console.log(city); // seoul

// Rest 프로퍼티 적용이 가능하다(마지막에 위치시켜야 함)
const {x, ...rest} = {x: 1, y: 2, z:3};
console.log(x, rest); // 1 {y:2, z: 3}
```
